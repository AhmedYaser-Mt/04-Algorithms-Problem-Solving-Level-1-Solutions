#include<iostream>
using namespace std;

int ReadPositiveNumber(string Message)
{
	int Number;
	do
	{
		cout << Message << endl;
		cin >> Number;

	} while (Number<0);
	return Number;
}

int Factorial(int N)
{
	int F = 1;

	for (int counter = N;counter >= 1;counter--)
	{
		F *= counter;
	}
	return F;
}

int main()
{
	cout << Factorial(ReadPositiveNumber("Enter positive number :")) << endl;
	return 0;
}
